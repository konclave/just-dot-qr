dot-qr — Architecture Design

Pipeline approach: pure core → scene model → pluggable renderers

Data Flow
user options
    ↓
generateMatrix(text)  ← delegates to qrcode/qr-code-generator
    ↓ boolean[][]
buildScene(matrix, options)  ← our core, pure function
    ↓ DotQRScene
SVGRenderer
→ string CanvasRenderer
→ void (draws to ctx) PNGRenderer
→ Blob / dataURL     ↑
ReactDotQR  ← wraps SVGRenderer as JSX
Core Types
interface DotQROptions {
  text: string
  size?: number                      // px, default 400
  dotColor?: string                  // default '#fff'
  backgroundColor?: string           // default 'transparent'
  dotScale?: number                  // 0–1, default 0.6
  finderStyle?: 'squares' | 'rounded' | 'circles'  // default 'circles'
  logo?: {
    src: string                        // URL or data URI
    width?: number                    // px
    height?: number                   // px, default = width
  }
}

interface DotQRScene {
  dots: Circle[]                      // data module circles
  finders: FinderShape[]             // 3 corner patterns
  size: number                       // canvas/svg dimensions
  backgroundColor: string
  logo?: LogoPlacement               // computed center rect
}
File Structure
src/
  index.ts               // public API exports
  types.ts               // DotQROptions, DotQRScene, etc.
  core/
    matrix.ts            // wraps QR library → boolean[][]
    scene.ts             // buildScene() — pure, DOM-free
    finder.ts            // finder pattern geometry
  renderers/
    svg.ts               // DotQRScene → SVG string
    canvas.ts            // DotQRScene → CanvasRenderingContext2D
    png.ts               // canvas → Blob / dataURL
  react/
    DotQR.tsx            // React component, peer dep
    index.ts             // re-exports DotQR component
Public API surface
// Core (framework-agnostic)
export { toSVG }        // (options) → string
export { toCanvas }     // (options, ctx) → void
export { toPNG }        // (options) → Promise<Blob>
export { buildScene }   // escape hatch for custom renderers
export type { DotQROptions, DotQRScene }

// React (from 'dot-qr/react')
export { DotQR }        // <DotQR text="..." size={400} />
